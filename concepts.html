printf and scanf

The printf function is used to display output on the console, while scanf is used to take user input. 


#include 
int main() {
    int num;
    printf("Enter a number: ");
    scanf("%d", &num);
    printf("You entered: %d\n", num);
    return 0;
}

Switch Case

The switch-case statement is used when multiple conditions need to be checked. 


#include 
int main() {
    int choice;
    printf("Enter a number (1-3): ");
    scanf("%d", &choice);
    switch (choice) {
        case 1: printf("You chose One\n"); break;
        case 2: printf("You chose Two\n"); break;
        case 3: printf("You chose Three\n"); break;
        default: printf("Invalid choice\n");
    }
    return 0;
}

Nested If

A nested if statement is an if statement inside another if statement. 


#include 
int main() {
    int num;
    printf("Enter a number: ");
    scanf("%d", &num);
    if (num > 0) {
        if (num % 2 == 0) {
            printf("Positive Even\n");
        } else {
            printf("Positive Odd\n");
        }
    } else {
        printf("Non-positive number\n");
    }
    return 0;
}

Loops

Loops in C include for, while, and do-while loops. 


#include 
int main() {
    for (int i = 1; i <= 5; i++) {
        printf("%d ", i);
    }
    return 0;
}

Functions

Functions are reusable blocks of code in C. 


#include 
void greet() {
    printf("Hello, Welcome to C Programming!\n");
}
int main() {
    greet();
    return 0;
}

Strings

Strings in C are arrays of characters ending with a null character '\0'. 


#include 
int main() {
    char name[20];
    printf("Enter your name: ");
    scanf("%s", name);
    printf("Hello, %s!\n", name);
    return 0;
}

Arrays

Arrays in C store multiple values of the same data type. 


#include 
int main() {
    int arr[5] = {1, 2, 3, 4, 5};
    for (int i = 0; i < 5; i++) {
        printf("%d ", arr[i]);
    }
    return 0;
}
